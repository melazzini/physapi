Detailed Description

template<>
class physapi::PCompontCollider< PMCCompton >

Template specialization of PCompontCollider for compton Monte Carlo simulation(PMCCompton).

See also
PMCCompton

Example:

#include <iostream>
#include "PCompontCollider.hpp"
#include "PMCCompton.hpp"
using namespace std;
using namespace physapi;
int main()
{
PCompontCollider<PMCCompton> comptonCollider;
PSimplePhoton photon{100.0E3_eV, PVector3D(1, 0, 0)};
cout << photon.energy() << endl;
cout << photon.momentum() << endl;
comptonCollider.run(photon, 100.0E3_eVToKelvin);
cout << photon.energy() << endl;
cout << photon.momentum() << endl;
}
