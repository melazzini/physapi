520  void makeLinEmptySpectrum()
521  {
522  /*
523 
524  when using a linear scale the energy of each interval is the mean:
525  E_i = (E_left_i + E_right_i)/2 = (E_left_i + E_left_i + dE)/2

544  void makeLogEmptySpectrum()
545  {
546  /*
547 
548  when using a logarithmic scale the energy of each interval is the mean:
549  E_i = (E_left_i + E_right_i)/2 = (E_left_i + E_left_i + dE)/2
550  = (2E_left_i + dE)/2
551 
552  E_i = E_left_i + (dE/2)
553 
554  where dE = E_left_{i+1} - E_left_i
555 
556  E_left_{i+1} = E_left * 10^{dExp*(index + 1)}
557  E_left_i = E_left * 10^{dExp*(index)}
558 
559  and
560 
561  dExp = (log(E_right) - log(E_left))/energyIntervals
562 
563  */
564  phys_float E_left_i = m_leftBound;
565 
566  for (size_t i = 0; i < m_spectrum.size(); i++)
567  {

569 
570  phys_float E_left_i = m_leftBound * std::pow(10, m_dE * i);
571 
572  auto dE = E_left_ip1 - E_left_i;
